344  std::vector< size_t > modelShapeGet( size_t inp_idx, size_t expected_size )
345  {
346  if( InputShape_exist( inp_idx ) && InputShape_get( inp_idx ).size() > 0 )
347  {
348  // InputShape has priority over InputN, InputH, InputW, InputC
349  auto shape = InputShape_get( inp_idx );
350  if( expected_size != 0 && shape.size() != expected_size )
351  DG_ERROR(
352  DG_FORMAT(
353  "The input shape parameter InputShape for input #"
354  << inp_idx << " must have " << expected_size << " elements, while it has " << shape.size() ),
355  ErrBadParameter );
356  return shape;
357  }
358  else
359  {
360  if( expected_size != 0 )
361  {
362  // when expected_size is set, we use it to fill the shape vector
363  // and put N/H/W/C values in the right places
364  std::vector< size_t > shape( expected_size, 1 );
365  if( InputN_get( inp_idx ) > 0 && expected_size >= 1 )
366  shape[ 0 ] = InputN_get( inp_idx );
367  if( InputH_get( inp_idx ) > 0 && expected_size >= 2 )
368  shape[ 1 ] = InputH_get( inp_idx );
369  if( InputW_get( inp_idx ) > 0 && expected_size >= 3 )
370  shape[ 2 ] = InputW_get( inp_idx );
371  if( InputC_get( inp_idx ) > 0 && expected_size >= 4 )
372  shape[ 3 ] = InputC_get( inp_idx );
373  return shape;
374  }
375  else
376  {
377  // when expected_size is not set, we fill the shape vector
378  // with all the values that are defined
379  std::vector< size_t > shape;
380  if( InputN_exist( inp_idx ) && InputN_get( inp_idx ) > 0 )
381  shape.push_back( (size_t)InputN_get( inp_idx ) );
382  if( InputH_exist( inp_idx ) && InputH_get( inp_idx ) > 0 )
383  shape.push_back( (size_t)InputH_get( inp_idx ) );
384  if( InputW_exist( inp_idx ) && InputW_get( inp_idx ) > 0 )
385  shape.push_back( (size_t)InputW_get( inp_idx ) );
386  if( InputC_exist( inp_idx ) && InputC_get( inp_idx ) > 0 )
387  shape.push_back( (size_t)InputC_get( inp_idx ) );
388  return shape;
389  }

474  template< typename T >
475  ModelParamsWriteAccess &paramSet( const char *section, const char *key, const T &value, size_t idx )
476  {
477  auto &param = ( section == nullptr || *section == '\0' ) ? m_cfg_rw : m_cfg_rw[ section ][ idx ];
478  if( !param.is_object() || !param.contains( key ) )
479  {
480  param[ key ] = value;
481  set_dirty( true );
482  }
483  else
484  {
485  typename std::remove_cv_t< std::remove_reference_t< decltype( value ) > > val;
486  auto &rec = param[ key ];
487  rec.get_to( val );
488  if( !rec.is_primitive() || rec.is_number_float() || val != value )
489  {
490  rec = value;
491  set_dirty( true );
492  }
493  }
494  return *this;
495  }
