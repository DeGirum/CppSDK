46  static inline bool jsonKeyExist( const json &json_params, const std::string &section, int index, const std::string &key )
47  {
48  const bool exists = section.empty() ? json_params.contains( key ) :
49  ( json_params.contains( section )
50  && json_params[ section ].is_array()
51  && index < json_params[ section ].size()
52  && json_params[ section ][ index ].contains( key ) );
53  return exists;
54  }

235  static BasicTensor tensorDeserialize( const json &j )
236  {
237  BasicTensor t;
238 
239  try
240  {
241  const int32_t id = j[ "id" ].get< int32_t >();
242  const std::string name = j[ "name" ].get< std::string >();
243  const BasicTensor::shape_t shape = j[ "shape" ].get< BasicTensor::shape_t >();
244  const auto q = j[ "quantization" ];
245  const int axis = q[ "axis" ].get< int >();
246  const std::vector< double > scales = q[ "scale" ].get< std::vector< double > >();
247  const std::vector< int64_t > zeros = q[ "zero" ].get< std::vector< int64_t > >();
248  const DGType type = String2DGType( j[ "type" ].get< std::string >().c_str() );
249 
250  size_t q_size = std::min( scales.size(), zeros.size() );
251  std::vector< BasicTensor::quant_params_t::scale_t > qparams( q_size );
252  for( size_t qi = 0; qi < q_size; qi++ )
253  {
254  qparams[ qi ].m_scale = scales[ qi ];
255  qparams[ qi ].m_zero = zeros[ qi ];
256  }
257 
258  const auto &byte_vector = j[ "data" ].get_binary();
259  t.alloc( id, name, shape, type, BasicTensor::quant_params_t( axis, qparams ) );
260  memcpy( t.untypedData(), byte_vector.data(), std::min( byte_vector.size(), t.linearSizeGet_bytes() ) );
261  }
262  catch( std::exception &e )
263  {
264  DG_ERROR( e.what(), ErrParseError );
265  }
266  return t;
267  }

275  {
