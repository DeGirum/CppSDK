78  static ServerAddress fromHostname( const std::string &hostname )
79  {
80  std::string pure_hostname = hostname;
81 
82  // deduce server protocol type
83  ServerType server_type = ServerType::ASIO;
84 
85  const std::vector< std::pair< std::string, ServerType > > prefixes =
86  { { "http://", ServerType::HTTP }, { "asio://", ServerType::ASIO } };
87 
88  for( const auto &prefix : prefixes )
89  {
90  if( const auto delim = pure_hostname.find( prefix.first ); delim != std::string::npos )
91  {
92  server_type = prefix.second;
93  pure_hostname = pure_hostname.substr( delim + prefix.first.length() );
94  break;
95  }
96  }
97 
98  // deduce TCP port
99  if( const auto delim = pure_hostname.rfind( ":" ); delim != std::string::npos )
100  return ServerAddress( pure_hostname.substr( 0, delim ), std::atoi( pure_hostname.substr( delim + 1 ).c_str() ), server_type );
101 
102  return ServerAddress( pure_hostname, DEFAULT_PORT, server_type );
103  }
104 
107  const bool is_valid() const
108  {
109  return !ip.empty();
110  }
111 
113  operator std::string() const
114  {
115  return ( server_type == ServerType::HTTP ? "http://" : "" ) + ip + ":" + std::to_string( port );
116  }
