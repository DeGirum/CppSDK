80  static ServerAddress fromHostname( const std::string &hostname )
81  {
82  std::string ip = hostname;
83 
84  // deduce server protocol type
85  ServerType server_type;
86  const std::string http_prefix = "http://";
87  if( const auto delim = ip.find( http_prefix ); delim != std::string::npos )
88  {
89  server_type = ServerType::HTTP;
90  ip = ip.substr( delim + http_prefix.length() );
91  }
92  else
93  server_type = ServerType::ASIO;
94 
95  // deduce TCP port
96  if( const auto delim = ip.rfind( ":" ); delim != std::string::npos )
97  return ServerAddress( ip.substr( 0, delim ), std::atoi( ip.substr( delim + 1 ).c_str() ), server_type );
98 
99  return ServerAddress( ip, DEFAULT_PORT, server_type );
100  }
101 
104  const bool is_valid() const
105  {
106  return !ip.empty();
107  }
108 
110  operator std::string() const
111  {
112  return ( server_type == ServerType::HTTP ? "http://" : "" ) + ip + ":" + std::to_string( port );
113  }
