576  void reshapeCombineDims( const std::vector< size_t > dims_to_combine )
577  {
578  // Check if dims_to_combine has at least 2 values
579  if( dims_to_combine.size() < 2 )
580  DG_ERROR( "reshapeCombineDims: dims_to_combine needs at least 2 dimensions to combine", ErrBadParameter );
581 
582  // Check if dims_to_combine has consecutive values
583  for( size_t i = 1; i < dims_to_combine.size(); i++ )
584  {
585  if( dims_to_combine[ i ] != dims_to_combine[ i - 1 ] + 1 )
586  DG_ERROR(
587  "reshapeCombineDims: dims_to_combine are not consecutive. Failed at index " + std::to_string( i ),
588  ErrBadParameter );
589  }
590 
591  // Check if the last value of dims_to_combine exceeds number of dimensions
592  if( dims_to_combine.back() >= m_shape.size() )
593  DG_ERROR( "reshapeCombineDims: dims_to_combine value exceeds tensor dimensions", ErrBadParameter );
594 
595  // Create new shape
596  size_t new_dim = m_shape.size() - dims_to_combine.size() + 1;
597  shape_t new_shape;
598  new_shape.reserve( new_dim );
599 
600  // Copy dimensions before the first dims_to_combine
601  for( size_t i = 0; i < dims_to_combine[ 0 ]; i++ )
602  new_shape.push_back( m_shape[ i ] );
603 
604  // Combine specified dimensions
605  size_t combined_dim = m_shape[ dims_to_combine[ 0 ] ];
606  for( size_t i = dims_to_combine[ 0 ] + 1; i <= dims_to_combine.back(); i++ )
607  combined_dim *= m_shape[ i ];
608  new_shape.push_back( combined_dim );
609 
610  // Copy remaining dimensions
611  for( size_t i = dims_to_combine.back() + 1; i < m_shape.size(); i++ )
612  new_shape.push_back( m_shape[ i ] );
613 
614  m_shape = new_shape;
615  }

657  template< typename T_IN, typename T_OUT >
658  void dequantize()
659  {
660  if( m_linear_size == 0 )
661  return;
662 
663  const T_IN *in = data< T_IN >();
664  if( in == nullptr )
665  DG_ERROR(
666  DG_FORMAT(
667  "Dequantize: tensor data type " << numpyTypeGet() << " does not match requested "
668  << NAMEOF_SHORT_TYPE( T_IN ) ),
669  ErrBadParameter );
670  T_OUT *out = new T_OUT[ m_linear_size ];
671 
672  const auto &qarr = m_quant_params.quant_params();
673  auto q_axis = m_quant_params.quant_axis();
674  size_t denom = 1; // linear index is divided by this value then...
675  size_t dim =
676  1; // ... the modulo by this value is calculated to obtain the index in quantization parameters array
677  if( q_axis >= 0 )
678  {
679  if( q_axis >= m_shape.size() )
680  DG_ERROR(
681  DG_FORMAT(
682  "Dequantize: tensor quantization axis " << q_axis << " is out of range 0.." << m_shape.size() ),
683  ErrBadParameter );
684  dim = m_shape[ q_axis ];
685  if( q_axis < m_shape.size() - 1 )
686  denom = std::accumulate( m_shape.begin() + q_axis + 1, m_shape.end(), 1, std::multiplies< size_t >() );
687  }
688 
689  for( size_t li = 0; li < m_linear_size; li++ )
690  {
691  const size_t qi = ( li / denom ) % dim;
692  const auto &qp = qarr[ qi ];
693  out[ li ] = (T_OUT)( ( (T_OUT)in[ li ] - qp.m_zero ) * qp.m_scale );
694  }
695 
696  if( !m_external )
697  delete[](char *)m_linear_buffer;
698  m_external = false;
699  m_linear_buffer = out;
700  m_el_size = sizeof( T_OUT );
701  m_type = &typeid( T_OUT );
702  }
