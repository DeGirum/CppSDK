235  template< typename T >

237  T ( ModelParamsReadAccess::*fallback )( size_t ) const ) const
238  {
239  if( paramExist( section, key, idx, &ModelParamsReadAccess::None_exist ) || fallback == &ModelParamsReadAccess::None_get< T > )

242  return jsonGetMandatoryValue< T >( m_cfg_ro, section, (int )idx, key );
243  else

335  template< typename T >
336  ModelParamsWriteAccess &paramSet( const char *section, const char *key, const T &value, size_t idx )
337  {
338  auto &param = ( section == nullptr || *section == '\0' ) ? m_cfg_rw : m_cfg_rw[ section ][ idx ];
339  if( !param.is_object() || !param.contains( key ) )
340  {
341  param[ key ] = value;
342  set_dirty( true );
343  }
344  else
345  {
346  typename std::remove_cv_t< std::remove_reference_t< decltype( value ) > > val;
347  auto &rec = param[ key ];
348  rec.get_to( val );
349  if( !rec.is_primitive() || rec.is_number_float() || val != value )
350  {
351  rec = value;
352  set_dirty( true );
353  }
354  }
355  return *this;
356  }
