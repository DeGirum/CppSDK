441  void reshapeToNHWC()
442  {
443  const size_t old_dim = m_shape.size();
444  const size_t new_dim = 4;
445  if( old_dim < new_dim )
446  {
447  m_shape.resize( new_dim, 1 );
448  if( old_dim == 2 )
449  std::swap( m_shape[ 3 ], m_shape[ 1 ] );
450  if( old_dim == 3 )

452  }
453  else if( old_dim > new_dim )
454  {
455  m_shape[ new_dim - 1 ] *= std::accumulate( m_shape.begin() + new_dim, m_shape.end(), (size_t )1, std::multiplies< size_t >() );

470  template < typename T_IN, typename T_OUT >
471  void dequantize()
472  {
473  if( m_linear_size == 0 )
474  return;
475 
476  const T_IN *in = data< T_IN >();
477  if( in == nullptr )
478  DG_ERROR(
479  DG_FORMAT( "Dequantize: tensor data type " << numpyTypeGet() << " does not match requested " << NAMEOF_SHORT_TYPE( T_IN ) ),
480  ErrBadParameter );
481  T_OUT *out = new T_OUT[ m_linear_size ];
482 
483  const auto &qarr = m_quant_params.quant_params();
484  auto q_axis = m_quant_params.quant_axis();
485  size_t denom = 1; // linear index is divided by this value then...
486  size_t dim = 1; // ... the modulo by this value is calculated to obtain the index in quantization parameters array
487  if( q_axis >= 0 )
488  {
489  if( q_axis >= m_shape.size() )
490  DG_ERROR(
491  DG_FORMAT( "Dequantize: tensor quantization axis " << q_axis << " is out of range 0.." << m_shape.size() ),
492  ErrBadParameter );
493  dim = m_shape[ q_axis ];
494  if( q_axis < m_shape.size() - 1 )
495  denom = std::accumulate( m_shape.begin() + q_axis + 1, m_shape.end(), 1, std::multiplies< size_t >() );
496  }
497 
498  for( size_t li = 0; li < m_linear_size; li++ )
499  {
500  const size_t qi = ( li / denom ) % dim;
501  const auto &qp = qarr[ qi ];
502  out[ li ] = (T_OUT)( ( (T_OUT)in[ li ] - qp.m_zero ) * qp.m_scale );
503  }
504 
505  if( !m_external )
506  delete[]( char * ) m_linear_buffer;
507  m_external = false;
508  m_linear_buffer = out;
509  m_el_size = sizeof( T_OUT );
510  m_type = &typeid( T_OUT );
511  }
